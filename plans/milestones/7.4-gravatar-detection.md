 * Using Web Crypto API (browser-compatible)

export async function md5Hash(input: string): Promise<string> {
  // Convert string to ArrayBuffer
  const encoder = new TextEncoder();
  const data = encoder.encode(input);

  // Use SubtleCrypto if available (modern browsers)
  if (typeof window !== 'undefined' && window.crypto?.subtle) {
    // Note: SubtleCrypto doesn't support MD5 directly
    // We'll use a pure JS implementation for compatibility
  }

  // Pure JavaScript MD5 implementation
  return md5(input);
}

/**
 * Pure JavaScript MD5 implementation
 * Based on RFC 1321

function md5(string: string): string {
  function rotateLeft(value: number, shift: number): number {
    return (value << shift) | (value >>> (32 - shift));
  }

  function addUnsigned(x: number, y: number): number {
    const lsw = (x & 0xffff) + (y & 0xffff);
    const msw = (x >> 16) + (y >> 16) + (lsw >> 16);
    return (msw << 16) | (lsw & 0xffff);
  }

  function f(x: number, y: number, z: number): number {
    return (x & y) | (~x & z);
  }
  function g(x: number, y: number, z: number): number {
    return (x & z) | (y & ~z);
  }
  function h(x: number, y: number, z: number): number {
    return x ^ y ^ z;
  }
  function i(x: number, y: number, z: number): number {
    return y ^ (x | ~z);
  }

  function ff(a: number, b: number, c: number, d: number, x: number, s: number, ac: number): number {
    a = addUnsigned(a, addUnsigned(addUnsigned(f(b, c, d), x), ac));
    return addUnsigned(rotateLeft(a, s), b);
  }
  function gg(a: number, b: number, c: number, d: number, x: number, s: number, ac: number): number {
    a = addUnsigned(a, addUnsigned(addUnsigned(g(b, c, d), x), ac));
    return addUnsigned(rotateLeft(a, s), b);
  }
  function hh(a: number, b: number, c: number, d: number, x: number, s: number, ac: number): number {
    a = addUnsigned(a, addUnsigned(addUnsigned(h(b, c, d), x), ac));
    return addUnsigned(rotateLeft(a, s), b);
  }
  function ii(a: number, b: number, c: number, d: number, x: number, s: number, ac: number): number {
    a = addUnsigned(a, addUnsigned(addUnsigned(i(b, c, d), x), ac));
    return addUnsigned(rotateLeft(a, s), b);
  }

  function convertToWordArray(str: string): number[] {
    const wordCount = ((str.length + 8) >> 6) + 1;
    const wordArray = new Array(wordCount * 16).fill(0);

    for (let i = 0; i < str.length; i++) {
      wordArray[i >> 2] |= str.charCodeAt(i) << ((i % 4) * 8);
    }

    wordArray[str.length >> 2] |= 0x80 << ((str.length % 4) * 8);
    wordArray[wordCount * 16 - 2] = str.length * 8;

    return wordArray;
  }

  function wordToHex(value: number): string {
    let hex = '';
    for (let i = 0; i <= 3; i++) {
      const byte = (value >> (i * 8)) & 255;
      hex += ('0' + byte.toString(16)).slice(-2);
    }
    return hex;
  }

  const x = convertToWordArray(string);
  let a = 0x67452301;
  let b = 0xefcdab89;
  let c = 0x98badcfe;
  let d = 0x10325476;

  const S21 = 5, S22 = 9, S23 = 14, S24 = 20;
  const S31 = 4, S32 = 11, S33 = 16, S34 = 23;
  const S41 = 6, S42 = 10, S43 = 15, S44 = 21;

    c = ff(c, d, a, b, x[k + 2], S13, 0x242070db);
    b = ff(b, c, d, a, x[k + 3], S14, 0xc1bdceee);
    a = ff(a, b, c, d, x[k + 4], S11, 0xf57c0faf);
    d = ff(d, a, b, c, x[k + 5], S12, 0x4787c62a);
    c = ff(c, d, a, b, x[k + 6], S13, 0xa8304613);
    b = ff(b, c, d, a, x[k + 7], S14, 0xfd469501);
    a = ff(a, b, c, d, x[k + 8], S11, 0x698098d8);
    d = ff(d, a, b, c, x[k + 9], S12, 0x8b44f7af);
    c = ff(c, d, a, b, x[k + 10], S13, 0xffff5bb1);
    b = ff(b, c, d, a, x[k + 11], S14, 0x895cd7be);

### Step 3: Create Gravatar Client

Create `src/lib/gravatar/client.ts`:

```typescript
import { GravatarResult, GravatarCheckResult } from './types';
import { md5Hash } from './hash';
import { LRUCache } from '../cache';

const GRAVATAR_BASE_URL = 'https://www.gravatar.com';

// Cache Gravatar results
const gravatarCache = new LRUCache<GravatarCheckResult>({
  maxSize: 500,
  ttlMs: 60 * 60 * 1000, // 1 hour
  name: 'gravatar',
});

export async function checkGravatar(email: string): Promise<GravatarCheckResult> {
  const normalizedEmail = email.trim().toLowerCase();

  // Check cache
  const cached = gravatarCache.get(normalizedEmail);
  if (cached) {
    return cached;
  }

  try {
    const hash = await md5Hash(normalizedEmail);

    // Check if Gravatar exists using d=404 parameter
    const checkUrl = `${GRAVATAR_BASE_URL}/avatar/${hash}?d=404&s=1`;

    const response = await fetch(checkUrl, {
      method: 'HEAD',
      signal: AbortSignal.timeout(5000),
    });

    if (response.ok) {
      // Gravatar exists
      const result: GravatarCheckResult = {
        checked: true,
        gravatar: {
          exists: true,
          hash,
          url: `${GRAVATAR_BASE_URL}/avatar/${hash}?s=200`,
          thumbnailUrl: `${GRAVATAR_BASE_URL}/avatar/${hash}?s=80`,
          profileUrl: `${GRAVATAR_BASE_URL}/${hash}`,
        },
        message: 'Gravatar profile found',
      };

      gravatarCache.set(normalizedEmail, result);
      return result;
    }

    // Gravatar doesn't exist
    const result: GravatarCheckResult = {
      checked: true,
      gravatar: {
        exists: false,
        hash,
        url: `${GRAVATAR_BASE_URL}/avatar/${hash}?d=mp`, // Default mystery person
        thumbnailUrl: `${GRAVATAR_BASE_URL}/avatar/${hash}?d=mp&s=80`,
      },
      message: 'No Gravatar profile',
    };

    gravatarCache.set(normalizedEmail, result);
    return result;
  } catch (error) {
    return {
      checked: false,
      message: error instanceof Error ? error.message : 'Failed to check Gravatar',
    };
  }
}

/**
 * Get Gravatar URL for an email (without checking if it exists)
 */
export async function getGravatarUrl(
  email: string,
  options: {
    size?: number;
    default?: 'mp' | 'identicon' | 'monsterid' | 'wavatar' | 'retro' | 'robohash' | 'blank';
  } = {}
): Promise<string> {
  const { size = 80, default: defaultImage = 'mp' } = options;
  const normalizedEmail = email.trim().toLowerCase();
  const hash = await md5Hash(normalizedEmail);

  return `${GRAVATAR_BASE_URL}/avatar/${hash}?s=${size}&d=${defaultImage}`;
}

export function clearGravatarCache(): void {
  gravatarCache.clear();
}
```

import { checkGravatar, GravatarCheckResult } from '../gravatar';

export async function checkGravatarProfile(
  email: string,
  options: { enabled?: boolean } = {}
): Promise<GravatarCheckResult> {
  const { enabled = true } = options;

  if (!enabled) {
    return {
      checked: false,
      message: 'Gravatar check disabled',
    };
  }

  return checkGravatar(email);
}

export { GravatarCheckResult };

describe('Gravatar', () => {

  });

    test('generates correct hash for email', async () => {
      // Known MD5 hash for "test@example.com"
      const hash = await md5Hash('test@example.com');
      expect(hash).toBe('55502f40dc8b7c769880b10874abc9d0');
    });

    test('generates consistent hash', async () => {
      const hash1 = await md5Hash('user@domain.com');
      const hash2 = await md5Hash('user@domain.com');
      expect(hash1).toBe(hash2);
    });

    test('generates different hash for different emails', async () => {
      const hash1 = await md5Hash('user1@domain.com');
      const hash2 = await md5Hash('user2@domain.com');
      expect(hash1).not.toBe(hash2);
    });
  });

  describe('checkGravatar', () => {
    test('returns exists: true when Gravatar found', async () => {
      (fetch as jest.Mock).mockResolvedValue({ ok: true });

      expect(result.checked).toBe(true);
      expect(result.gravatar?.exists).toBe(true);
      expect(result.gravatar?.url).toContain('gravatar.com/avatar');
    });

    test('returns exists: false when Gravatar not found', async () => {
      (fetch as jest.Mock).mockResolvedValue({ ok: false, status: 404 });
